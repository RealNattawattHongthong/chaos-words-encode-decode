```python
def add_padding(s):
    """‡πÄ‡∏ï‡∏¥‡∏° = ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ‡∏Ñ‡∏ß‡∏≤‡∏°‡∏¢‡∏≤‡∏ß‡∏´‡∏≤‡∏£‡∏î‡πâ‡∏ß‡∏¢ 4 ‡∏•‡∏á‡∏ï‡∏±‡∏ß"""
    missing = len(s) % 4
    if missing:
        return s + "=" * (4 - missing)
    return s
```
